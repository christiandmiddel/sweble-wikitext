"\n" * n "Hallo"

n
0 -> <Hallo>
1 -> <Hallo>

2 -> <BR Hallo>            1
3 -> <BR><Hallo>           2
4 -> <BR><BR Hallo>        3
5 -> <BR><BR><Hallo>       4
6 -> <BR><BR><BR Hallo>    5
7 -> <BR><BR><BR><Hallo>   6

gap = max(0, n - 1)

<BR> * gap/2 <BR * gap%2 Hallo>
